<body>

    <div id="game-container">
        <div id="board"></div>
    </div>

    <div id="hud">
        <h1>Chess RPG</h1>
        <div class="stat">HP: <span id="hp">20</span></div>
        <div class="stat">Attack: <span id="atk">5</span></div>
        <div class="stat">Position: <span id="pos">0,0</span></div>
        <p>Controls: W/A/S/D to move (like a chess king), Space to attack.</p>
        <div id="log"></div>
    </div>

    <script>
        const boardEl = document.getElementById("board");
        const hpEl = document.getElementById("hp");
        const atkEl = document.getElementById("atk");
        const posEl = document.getElementById("pos");
        const logEl = document.getElementById("log");

        const BOARD_SIZE = 8;

        // Player stats
        let player = {
            x: 0,
            y: 7, // bottom-left
            hp: 20,
            atk: 5
        };

        // Simple enemy
        let enemy = {
            x: 4,
            y: 3,
            hp: 10
        };

        // Create board tiles
        let tiles = [];
        for (let y = 0; y < BOARD_SIZE; y++) {
            for (let x = 0; x < BOARD_SIZE; x++) {
                const tile = document.createElement("div");
                tile.classList.add("tile");
                if ((x + y) % 2 === 0) {
                    tile.classList.add("light");
                } else {
                    tile.classList.add("dark");
                }
                tile.dataset.x = x;
                tile.dataset.y = y;
                boardEl.appendChild(tile);
                tiles.push(tile);
            }
        }

        function getTile(x, y) {
            return tiles.find(t => Number(t.dataset.x) === x && Number(t.dataset.y) === y);
        }

        function render() {
            // Clear pieces
            tiles.forEach(t => {
                t.innerHTML = "";
            });

            // Draw player
            const playerTile = getTile(player.x, player.y);
            const playerDiv = document.createElement("div");
            playerDiv.classList.add("player");
            playerTile.appendChild(playerDiv);

            // Draw enemy if alive
            if (enemy.hp > 0) {
                const enemyTile = getTile(enemy.x, enemy.y);
                const enemyDiv = document.createElement("div");
                enemyDiv.classList.add("enemy");
                enemyTile.appendChild(enemyDiv);
            }

            // Update HUD
            hpEl.textContent = player.hp;
            atkEl.textContent = player.atk;
            posEl.textContent = player.x + "," + player.y;
        }

        function log(message) {
            const p = document.createElement("p");
            p.textContent = message;
            logEl.appendChild(p);
            logEl.scrollTop = logEl.scrollHeight;
        }

        function movePlayer(dx, dy) {
            const newX = player.x + dx;
            const newY = player.y + dy;

            // Stay on board
            if (newX < 0 || newX >= BOARD_SIZE || newY < 0 || newY >= BOARD_SIZE) {
                log("You can't move off the board.");
                return;
            }

            player.x = newX;
            player.y = newY;
            log("You moved to " + newX + "," + newY);
            render();
        }

        function attack() {
            if (player.x === enemy.x && player.y === enemy.y && enemy.hp > 0) {
                enemy.hp -= player.atk;
                log("You attacked the enemy! Enemy HP: " + enemy.hp);
                if (enemy.hp <= 0) {
                    log("Enemy defeated!");
                }
            } else {
                log("No enemy on this tile.");
            }
            render();
        }

        // Keyboard controls: W/A/S/D + Space
        document.addEventListener("keydown", (e) => {
            if (e.key === "w") movePlayer(0, -1);
            if (e.key === "s") movePlayer(0, 1);
            if (e.key === "a") movePlayer(-1, 0);
            if (e.key === "d") movePlayer(1, 0);
            if (e.key === " ") attack();
        });

        // Initial render
        render();
        log("Welcome to Chess RPG. Move to the red enemy and press Space to attack.");
    </script>
